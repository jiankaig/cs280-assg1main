576char* ObjectAllocator::CreateAPage(){
577 try{
578 //1. Request memory for first page
579 NewPage = new char[stats_.PageSize_];
580 GenericObject* oldPage = PageList_;
581 //2. Casting the page to a GenericObject* and adjusting pointers.

590 GenericObject* currentObj = NULL;

592 char* NewObject = NewPage + sizeof(void*) + config_.PadBytes_
593 + config_.HBlockInfo_.size_ + config_.LeftAlignSize_; // point after page's Next
594 memset(NewObject-config_.PadBytes_-config_.HBlockInfo_.size_, 0x00, config_.HBlockInfo_.size_);
595 memset(NewObject-config_.PadBytes_, PAD_PATTERN, config_.PadBytes_);
596 memset(NewObject, UNALLOCATED_PATTERN, stats_.ObjectSize_);
597 memset(NewObject+stats_.ObjectSize_, PAD_PATTERN, config_.PadBytes_);

600 FreeList_->Next = NULL;
601 currentObj = NULL;
602 NewObject = NULL;
603

606 while(countBlocks){
607 currentObj = FreeList_;
608 NewObject = reinterpret_cast<char* >(FreeList_);

641int ObjectAllocator::FindPageByObject(GenericObject* &p, void* Object) const{
642 GenericObject* page = reinterpret_cast<GenericObject*>(p);
643 size_t ObjLowBou;
644 size_t ObjUppBou;
645 while(page){
646 // find right page..
647 ObjLowBou = reinterpret_cast<size_t>(Object) - reinterpret_cast<size_t>(page);
648 ObjUppBou = reinterpret_cast<size_t>(page) + stats_.PageSize_ - reinterpret_cast<size_t>(Object);
649
650 if(ObjLowBou + ObjUppBou == stats_.PageSize_ && ObjLowBou < stats_.PageSize_){
651 // found right page
652 p = page;
653 return 1;
654 }
655
656 // flip to next page
657 page = page->Next;
658 }
659 return -1; // page not found
660}
