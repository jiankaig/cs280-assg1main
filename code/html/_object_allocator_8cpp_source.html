106void* ObjectAllocator::Allocate(const char *label) {
107
108 if(config_.UseCPPMemManager_){
109 stats_.Allocations_ += 1;
110 size_t size = stats_.ObjectSize_;
111
112 if(stats_.Allocations_ > stats_.MostObjects_ )
113 stats_.MostObjects_ = stats_.Allocations_;
114

116 }
117
118
119 if (stats_.FreeObjects_ == 0 && stats_.PagesInUse_ < config_.MaxPages_){
120 //check page limit before adding new page

123 if(newPageStart < lowerBoundary)
124 lowerBoundary = newPageStart;
125
126 newPageStart += stats_.PageSize_;
127 if(newPageStart > upperBoundary)
128 upperBoundary = newPageStart;
129
130 }

132

203void ObjectAllocator::Free(void *Object){
204
205 if(config_.UseCPPMemManager_){
206 stats_.Deallocations_ += 1;
207 delete [] reinterpret_cast<char*>(Object);
208 return;
209 }
210
211 //check for "double free"
212 bool bNoDoubleFree = true;
213 GenericObject* ptrFreeList = nullptr;
214 ptrFreeList = FreeList_;
215
216 while(ptrFreeList){
217 if(ptrFreeList == reinterpret_cast<GenericObject* >(Object)){
218 bNoDoubleFree = false; // repeated object freed
219 break;
220 }
221 ptrFreeList = ptrFreeList->Next;
222 }
223 ptrFreeList = nullptr;
224
225 //check for out-of-bounds
226 if(Object < lowerBoundary || Object > upperBoundary)
227 {
228 throw OAException(OAException::E_BAD_BOUNDARY, "bad boundary. ");
229 }
230
231 //check if page can be found
232 GenericObject* page = PageList_;
233 int ret = FindPageByObject(page, Object);
234 if(ret == -1)
235 throw OAException(OAException::E_BAD_BOUNDARY, "bad boundary. page not found ");
236
237 //check if object is aligned and base case
238 size_t ObjectPosition = (size_t)Object - (size_t)page;
239 // printf("OP:%p = Obj: %p - pg: %p\n",(void*)ObjectPosition, Object, (void*)page);
240 bool isAligned = ((ObjectPosition - 8 - config_.PadBytes_ - config_.HBlockInfo_.size_)
241 % (stats_.ObjectSize_ + 2*config_.PadBytes_ + config_.HBlockInfo_.size_)) == 0;
242 bool baseCase = ObjectPosition % stats_.PageSize_ == 0; //special case
243 if(!isAligned || baseCase){
244 throw OAException(OAException::E_BAD_BOUNDARY, "bad boundary. mis-alignment ");
245 }
246
247 //check left and right pad bytes for pad pattern
248 if(config_.PadBytes_ != 0){
249 PaddState ps = isPadCorrupted(Object);
250 if(ps == CORRUPT_LEFT)
251 throw OAException(OAException::E_CORRUPTED_BLOCK,"corruption on left");
252 if(ps == CORRUPT_RIGHT)
253 throw OAException(OAException::E_CORRUPTED_BLOCK,"corruption on right");
254
255 }
256
257
258
259 //Free Object if no issues
260 if(bNoDoubleFree){
261 GenericObject* temp = FreeList_;
262 FreeList_ = reinterpret_cast<GenericObject* >(Object);
263 FreeList_->Next = temp;
264 temp = NULL;
265
266 char* ptrToHeaderBlock = reinterpret_cast<char* >(Object)-config_.PadBytes_ - config_.HBlockInfo_.size_;
267 char* ptrToAllocNum = NULL;
268 size_t BytesInBasicBlock;
269 if(config_.HBlockInfo_.type_ == OAConfig::hbExtended){
270 ptrToAllocNum = ptrToHeaderBlock + config_.HBlockInfo_.additional_ + 2;
271 BytesInBasicBlock = config_.HBlockInfo_.size_-config_.HBlockInfo_.additional_-2;
272 }
273 else{
274 ptrToAllocNum = ptrToHeaderBlock;
275 BytesInBasicBlock = config_.HBlockInfo_.size_;
276 }
277 memset(ptrToAllocNum, ZERO_PATTERN, BytesInBasicBlock);
278 char* ptrToFreedPatternArea = reinterpret_cast<char*>(FreeList_) + sizeof(void*);
279 memset(ptrToFreedPatternArea, FREED_PATTERN, stats_.ObjectSize_-sizeof(void*));
280
281
282 //update acounting info
283 stats_.ObjectsInUse_ -= 1;
284 stats_.FreeObjects_ += 1;
285 stats_.Deallocations_ += 1;
286 }
287 else if(!bNoDoubleFree){
288 throw OAException(OAException::E_MULTIPLE_FREE, "double free. ");
289 }
290
291}

473char* ObjectAllocator::CreateAPage(){
474 try{
475 //1. Request memory for first page
476 NewPage = new char[stats_.PageSize_];
477 GenericObject* oldPage = PageList_;
478 //2. Casting the page to a GenericObject* and adjusting pointers.
479 PageList_ = reinterpret_cast<GenericObject* >(NewPage);
480 PageList_->Next = oldPage;
481 oldPage = NULL;
482 stats_.PagesInUse_ += 1;//update acounting info
483 stats_.FreeObjects_ += config_.ObjectsPerPage_;//update acounting info
484
485
486 //3. Casting the 1st 16-byte block to GenericObject* and putting on free list
487 GenericObject* currentObj = NULL;
488 currentObj = FreeList_;
489 char* NewObject = NewPage + sizeof(size_t) + config_.PadBytes_ + config_.HBlockInfo_.size_; // point after page's Next
490 memset(NewObject-config_.PadBytes_-config_.HBlockInfo_.size_, 0x00, config_.HBlockInfo_.size_);
491 memset(NewObject-config_.PadBytes_, PAD_PATTERN, config_.PadBytes_);
492 memset(NewObject, UNALLOCATED_PATTERN, stats_.ObjectSize_);
493 memset(NewObject+stats_.ObjectSize_, PAD_PATTERN, config_.PadBytes_);
494 FreeList_ = reinterpret_cast<GenericObject*>(NewObject);
495 FreeList_->Next = NULL;
496 currentObj = NULL;
497 NewObject = NULL;
498
499 // cast remaining blocks
500 unsigned int countBlocks = config_.ObjectsPerPage_ - 1;
501 while(countBlocks){
502 currentObj = FreeList_;
503 NewObject = reinterpret_cast<char* >(FreeList_);
504 NewObject = NewObject + stats_.ObjectSize_ + config_.PadBytes_*2 + config_.HBlockInfo_.size_;
505 memset(NewObject-config_.PadBytes_-config_.HBlockInfo_.size_, 0x00, config_.HBlockInfo_.size_);
506 memset(NewObject-config_.PadBytes_, PAD_PATTERN, config_.PadBytes_);
507 memset(NewObject, UNALLOCATED_PATTERN, stats_.ObjectSize_);
508 memset(NewObject+stats_.ObjectSize_, PAD_PATTERN, config_.PadBytes_);
509 FreeList_ = reinterpret_cast<GenericObject* >(NewObject);
510 FreeList_->Next = currentObj;
511 currentObj = NULL;
512 NewObject = NULL;
513
514 countBlocks--;
515 }
516
517 }
518 catch(std::bad_alloc &){
519 throw OAException(OAException::E_NO_MEMORY, "allocate_new_page: No system memory available.");
520 }
521 return NewPage;
522}

535int ObjectAllocator::FindPageByObject(GenericObject* &p, void* Object) const{
536 GenericObject* page = reinterpret_cast<GenericObject*>(p);
537 size_t ObjLowBou;
538 size_t ObjUppBou;
539 while(page){
540 // find right page..
541 ObjLowBou = reinterpret_cast<size_t>(Object) - reinterpret_cast<size_t>(page);
542 ObjUppBou = reinterpret_cast<size_t>(page) + stats_.PageSize_ - reinterpret_cast<size_t>(Object);
543
544 if(ObjLowBou + ObjUppBou == stats_.PageSize_ && ObjLowBou < stats_.PageSize_){
545 // found right page
546 p = page;
547 return 1;
548 }
549
550 // flip to next page
551 page = page->Next;
552 }
553 return -1; // page not found
554}

565ObjectAllocator::PaddState ObjectAllocator::isPadCorrupted(void* ptrToBlock) const{
566 //check left and right pad bytes for pad pattern
567 if(config_.PadBytes_ != 0){
568 char* leftPadStart = reinterpret_cast<char*>(ptrToBlock) - config_.PadBytes_;
569 char* rightPadStart = reinterpret_cast<char*>(ptrToBlock) + stats_.ObjectSize_;
570
571 int count = config_.PadBytes_; // check each byte indiviually
572 while(count!=0){
573 if(static_cast<unsigned char>(*leftPadStart)!=PAD_PATTERN){
574 return ObjectAllocator::CORRUPT_LEFT; //corruptioon found
575 }
576 if(static_cast<unsigned char>(*rightPadStart)!= PAD_PATTERN){
577 return ObjectAllocator::CORRUPT_RIGHT; //corruptioon found
578 }
579 ptrToBlock = reinterpret_cast<char*>(ptrToBlock) +1;
580 count--;
581 }
582
583 }
584 return ObjectAllocator::NO_CORRUPT; // no corruption
585}
