131 struct HeaderBlockInfo

147 HeaderBlockInfo(HBLOCK_TYPE type = hbNone, unsigned additional = 0) : type_(type), size_(0), additional_(additional)
148 {
149 if (type_ == hbBasic)
150 size_ = BASIC_HEADER_SIZE;
151 else if (type_ == hbExtended) // alloc # + use counter + flag byte + user-defined
152 size_ = sizeof(unsigned int) + sizeof(unsigned short) + sizeof(char) + additional_;
153 else if (type_ == hbExternal)
154 size_ = EXTERNAL_HEADER_SIZE;
155 };
156 };
