143 HeaderBlockInfo(HBLOCK_TYPE type = hbNone, unsigned additional = 0) : type_(type), size_(0), additional_(additional)
144 {
145 if (type_ == hbBasic)
146 size_ = BASIC_HEADER_SIZE;
147 else if (type_ == hbExtended) // alloc # + use counter + flag byte + user-defined

151 };
